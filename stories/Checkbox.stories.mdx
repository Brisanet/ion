import { Canvas, Meta, Story } from '@storybook/addon-docs';
import { CheckboxComponent } from '../projects/ion/src/lib/checkbox/checkbox.component.ts';

<Meta title="Ion/Data Entry/Checkbox" component={CheckboxComponent} />

export const Template = (args) => <CheckboxComponent {...args} />;

# Checkbox

No Ion o `checkbox` pode receber os paramêtros `disabled` e `state`

<Canvas>
  <Story name="type: default">{Template.bind({})}</Story>
</Canvas>

## Estado(state)

O paramêtro `state` serve para que o usuário especificar um estado para o checkbox, podendo receber:

- `enabled`(valor padrão)
- `checked`
- `indeterminate`

<Canvas>
  <Story name="type: enabled" args={{ state: 'enabled' }}>
    {Template.bind({})}
  </Story>
  <Story name="type: checked" args={{ state: 'checked' }}>
    {Template.bind({})}
  </Story>
  <Story name="type: indeterminate" args={{ state: 'indeterminate' }}>
    {Template.bind({})}
  </Story>
</Canvas>

## Disabled

Caso o parametro `disabled` seja passado como verdadeiro o `checkbox` será rederizado desabilitado

<Canvas>
  <Story name="type: disabled" args={{ disabled: true }}>
    {Template.bind({})}
  </Story>
</Canvas>

O mesmo pode ser usado em conjunto com o paramêtro `state`.

<Canvas>
  <Story
    name="type: disabled-checked"
    args={{ state: 'checked', disabled: true }}
  >
    {Template.bind({})}
  </Story>
  <Story
    name="type: disabled-indeterminate"
    args={{ state: 'indeterminate', disabled: true }}
  >
    {Template.bind({})}
  </Story>
</Canvas>

## With label

Caso um label seja passado o `checkbox` será rederizado com o mesmo.

<Canvas>
  <Story name="type: with label" args={{ label: 'Label' }}>
    {Template.bind({})}
  </Story>
</Canvas>
